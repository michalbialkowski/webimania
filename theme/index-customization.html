@mixin button-variant(
    $background,
    $border,
    $color: $white,
    $hover-background: tint-color($background, $btn-hover-bg-tint-amount),
    $hover-color: $white,
    $active-background: tint-color($background, $btn-active-bg-tint-amount),
    $active-border: if($color == $color-contrast-light, shade-color($border, $btn-active-border-shade-amount), tint-color($border, $btn-active-border-tint-amount)),
    $active-color: $white,
    $disabled-background: lightness($background),
    $disabled-color: darkness($background)
) {
    color: $color;
    @include gradient-bg($background);
    border-color: transparent;
    box-shadow: none;
    
    &.btn-light {
        color: lighten($dark, 25%);
    }

    &:hover {
        color: $hover-color;
        &.btn-light {
            color: lighten($dark, 25%);
        }
        @include gradient-bg($hover-background);
    }

    .btn-check:focus + &,
    &:focus {
        color: $hover-color;
        &.btn-light {
            color: lighten($dark, 25%);
        }
        @include gradient-bg($hover-background);
        @if $enable-shadows {
            box-shadow: none;
        } @else {
            box-shadow: none;
        }
    }

    .btn-check:checked + &,
    .btn-check:active + &,
    &:active,
    &.active,
    .show > &.dropdown-toggle {
        color: $color;
        &.btn-light {
            color: lighten($dark, 25%);
        }
        background-color: $active-background;
        // Remove CSS gradients if they're enabled
        background-image: if($enable-gradients, none, null);

        &:focus {
            @if $enable-shadows {
                box-shadow: none;
            } @else {
                box-shadow: none;
            }
        }
    }

    &:disabled,
    &.disabled {
        color: $disabled-color;
        background-color: $disabled-background;
        // Remove CSS gradients if they're enabled
        background-image: if($enable-gradients, none, null);
        border-color: transparent;
    }
}
